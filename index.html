Google Sheets Setup Guide

  1. Open Google Sheets → create a new spreadsheet.
  2. Rename Sheet1 → Inventory → add second sheet named Invoices.
  3. Go to Extensions → Apps Script.
  4. Delete default code → paste the script below.
  5. Click Deploy → New deployment → Web app.
  6. Set Execute as: Me  |  Who has access: Anyone.
  7. Authorize → copy the Web app URL → paste it in the app.
function doGet(e) {
  var action = e.parameter.action;
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  if (action === "pullInventory") {
    var sheet = ss.getSheetByName("Inventory");
    if (!sheet || sheet.getLastRow() < 2) return out({ok:true,items:[]});
    var rows = sheet.getRange(2, 1, sheet.getLastRow()-1, 5).getValues();
    var items = rows.filter(r=>r[0]).map(r=>({id:Number(r[0]),name:r[1],qty:Number(r[2]),unit:r[3],price:Number(r[4])}));
    return out({ok:true,items:items});
  }

  if (action === "pullInvoices") {
    var sheet = ss.getSheetByName("Invoices");
    if (!sheet || sheet.getLastRow() < 2) return out({ok:true,invoices:[]});
    var rows = sheet.getRange(2, 1, sheet.getLastRow()-1, 7).getValues();
    var invs = rows.filter(r=>r[0]).map(r=>{
      var items=[]; try{items=JSON.parse(r[6]);}catch(e){}
      var ts = r[3]; if(ts instanceof Date){ts=ts.toISOString();}
      return {invNo:String(r[0]),date:String(r[1]),time:String(r[2]),timestamp:ts,total:Number(r[4]),items:items};
    });
    return out({ok:true,invoices:invs});
  }

  return out({ok:false,error:"Unknown action"});
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var body = JSON.parse(e.postData.contents);

  if (body.action === "pushInventory") {
    var sheet = ss.getSheetByName("Inventory") || ss.insertSheet("Inventory");
    // Ensure header row exists
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["id","name","qty","unit","price"]);
    }

    var changed = body.data.changed || [];
    var deleted = body.data.deleted || [];

    // ── UPSERT changed items ──────────────────────────────────
    changed.forEach(function(item) {
      if (sheet.getLastRow() < 2) {
        // Sheet has only header — just append
        sheet.appendRow([item.id, item.name, item.qty, item.unit, item.price]);
        return;
      }
      var ids = sheet.getRange(2, 1, sheet.getLastRow()-1, 1).getValues();
      var found = false;
      for (var r = 0; r < ids.length; r++) {
        if (Number(ids[r][0]) === item.id) {
          sheet.getRange(r+2, 1, 1, 5).setValues([[item.id, item.name, item.qty, item.unit, item.price]]);
          found = true; break;
        }
      }
      if (!found) sheet.appendRow([item.id, item.name, item.qty, item.unit, item.price]);
    });

    // ── DELETE removed items (reverse order to avoid row-shift bug) ──
    if (deleted.length > 0 && sheet.getLastRow() >= 2) {
      var ids = sheet.getRange(2, 1, sheet.getLastRow()-1, 1).getValues();
      // Collect row numbers to delete (1-indexed sheet rows)
      var rowsToDelete = [];
      deleted.forEach(function(id) {
        for (var r = 0; r < ids.length; r++) {
          if (Number(ids[r][0]) === id) { rowsToDelete.push(r+2); break; }
        }
      });
      // Sort descending — delete from bottom up so row numbers don't shift
      rowsToDelete.sort(function(a,b){return b-a;});
      rowsToDelete.forEach(function(row){ sheet.deleteRow(row); });
    }

    return out({ok:true});
  }

  if (body.action === "logInvoice") {
    var sheet = ss.getSheetByName("Invoices") || ss.insertSheet("Invoices");
    if (sheet.getLastRow() === 0) sheet.appendRow(["invNo","date","time","timestamp","total","itemCount","items"]);
    var inv = body.data;
    sheet.appendRow([inv.invNo, inv.date, inv.time, inv.timestamp, inv.total, inv.items.length, JSON.stringify(inv.items)]);
    return out({ok:true});
  }

  return out({ok:false,error:"Unknown action"});
}

function out(obj){
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

⚠️ The script uses a diff-based strategy — only changed and deleted rows are written. Deletes use reverse-order to avoid row-shift bugs. After any script change, create a New Deployment (not update existing).
