Google Sheets Setup Guide

  1. Open Google Sheets and create a new spreadsheet.
  2. Rename Sheet1 → Inventory and create a second sheet named Invoices.
  3. Go to Extensions → Apps Script.
  4. Delete any default code and paste the script below.
  5. Click Deploy → New deployment → Web app.
  6. Set Execute as: Me  |  Who has access: Anyone.
  7. Authorize and copy the Web app URL.
  8. Paste the URL into the Sheets URL field in this app.
function doGet(e) {
  var action = e.parameter.action;
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  if (action === "pullInventory") {
    var sheet = ss.getSheetByName("Inventory");
    if (!sheet || sheet.getLastRow() < 2) return ContentService.createTextOutput(JSON.stringify({ok:true,items:[]})).setMimeType(ContentService.MimeType.JSON);
    var rows = sheet.getRange(2, 1, sheet.getLastRow()-1, 5).getValues();
    var items = rows.filter(r=>r[0]).map(r=>({id:Number(r[0]),name:r[1],qty:Number(r[2]),unit:r[3],price:Number(r[4])}));
    return ContentService.createTextOutput(JSON.stringify({ok:true,items:items})).setMimeType(ContentService.MimeType.JSON);
  }

  if (action === "pullInvoices") {
    var sheet = ss.getSheetByName("Invoices");
    if (!sheet || sheet.getLastRow() < 2) return ContentService.createTextOutput(JSON.stringify({ok:true,invoices:[]})).setMimeType(ContentService.MimeType.JSON);
    var rows = sheet.getRange(2, 1, sheet.getLastRow()-1, 7).getValues();
    var period = e.parameter.period || "all";
    var now = new Date();
    var todayStart = new Date(now.getFullYear(), now.getMonth(), now.getDate());
    var weekStart  = new Date(todayStart); weekStart.setDate(weekStart.getDate() - weekStart.getDay());
    var monthStart = new Date(now.getFullYear(), now.getMonth(), 1);
    var invs = rows.filter(function(r){ return r[0]; }).map(function(r) {
      var items=[]; try{items=JSON.parse(r[6]);}catch(e){}
      var ts = r[3];
      if (ts instanceof Date) { ts = ts.toISOString(); }
      else if (typeof ts === "number") { ts = new Date((ts - 25569) * 86400 * 1000).toISOString(); }
      else { ts = String(ts); }
      return {invNo:String(r[0]),date:String(r[1]),time:String(r[2]),timestamp:ts,total:Number(r[4]),items:items};
    }).filter(function(inv) {
      if (period === "all") return true;
      var d = inv.timestamp ? new Date(inv.timestamp) : null;
      if (!d || isNaN(d.getTime())) return false;
      if (period === "today") return d >= todayStart;
      if (period === "week")  return d >= weekStart;
      if (period === "month") return d >= monthStart;
      return true;
    });
    return ContentService.createTextOutput(JSON.stringify({ok:true,invoices:invs})).setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({ok:false,error:"Unknown action"})).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var body = JSON.parse(e.postData.contents);
  var action = body.action;

  if (action === "pushInventory") {
    var sheet = ss.getSheetByName("Inventory") || ss.insertSheet("Inventory");
    var all = body.data.all;
    var changed = body.data.changed;
    var deleted = body.data.deleted;
    if (sheet.getLastRow() < 2) {
      sheet.clearContents();
      sheet.appendRow(["id","name","qty","unit","price"]);
      all.forEach(function(i){ sheet.appendRow([i.id,i.name,i.qty,i.unit,i.price]); });
      return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    }
    var rows = sheet.getRange(2, 1, sheet.getLastRow()-1, 1).getValues();
    changed.forEach(function(item) {
      for (var r = 0; r < rows.length; r++) {
        if (Number(rows[r][0]) === item.id) {
          sheet.getRange(r+2, 1, 1, 5).setValues([[item.id, item.name, item.qty, item.unit, item.price]]);
          return;
        }
      }
      sheet.appendRow([item.id, item.name, item.qty, item.unit, item.price]);
    });
    deleted.forEach(function(id) {
      var currentRows = sheet.getRange(2, 1, sheet.getLastRow()-1, 1).getValues();
      for (var r = 0; r < currentRows.length; r++) {
        if (Number(currentRows[r][0]) === id) { sheet.deleteRow(r+2); return; }
      }
    });
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }

  if (action === "logInvoice") {
    var sheet = ss.getSheetByName("Invoices") || ss.insertSheet("Invoices");
    if (sheet.getLastRow() === 0) sheet.appendRow(["invNo","date","time","timestamp","total","itemCount","items"]);
    var inv = body.data;
    sheet.appendRow([inv.invNo, inv.date, inv.time, String(inv.timestamp), inv.total, inv.items.length, JSON.stringify(inv.items)]);
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({ok:false,error:"Unknown action"})).setMimeType(ContentService.MimeType.JSON);
}

function backupData() {
  var BACKUP_SHEET_ID = "PASTE_YOUR_BACKUP_SPREADSHEET_ID_HERE";
  var source = SpreadsheetApp.getActiveSpreadsheet();
  var dest   = SpreadsheetApp.openById(BACKUP_SHEET_ID);
  var timestamp = Utilities.formatDate(new Date(), "Asia/Kolkata", "yyyy-MM-dd");
  var sheetsToCopy = ["Inventory", "Invoices"];
  sheetsToCopy.forEach(function(name) {
    var src = source.getSheetByName(name);
    if (!src) return;
    var backupName = name + "_backup_" + timestamp;
    var existing = dest.getSheetByName(backupName);
    if (existing) dest.deleteSheet(existing);
    var copy = src.copyTo(dest);
    copy.setName(backupName);
    var allBackups = dest.getSheets().filter(function(s) {
      return s.getName().startsWith(name + "_backup_");
    }).sort(function(a, b) { return a.getName().localeCompare(b.getName()); });
    while (allBackups.length > 30) dest.deleteSheet(allBackups.shift());
  });
  Logger.log("Backup done: " + timestamp);
}

⚠️ After any changes to the script, always create a New Deployment (not update existing) to avoid caching issues.
